sequenceDiagram
    autonumber
    actor Part as Participant
    participant View as Event Detail (Blade)
    participant Ctrl as RegistrationController
    participant DB as MongoDB (events & registrations)

    Part->>View: Klik Tombol "Daftar Sekarang"
    activate View
    View->>Ctrl: HTTP POST /events/{id}/register
    deactivate View
    activate Ctrl
    
    Ctrl->>DB: Count Registrations where event_id = ID
    activate DB
    DB-->>Ctrl: Kembalikan Jumlah Pendaftar Saat Ini
    
    Ctrl->>DB: Fetch Detail Event (Ambil nilai 'kapasitas')
    DB-->>Ctrl: Kembalikan Data Event
    deactivate DB

    alt Jumlah Pendaftar >= Kapasitas Event
        Ctrl-->>Part: Redirect Back dengan Error "Kuota Sudah Penuh!"
    else Kuota Masih Tersedia
        Ctrl->>Ctrl: Generate Unique registration_code (EVT-XXXX)
        Ctrl->>DB: Insert Dokumen ke Koleksi 'registrations' (Status: Registered)
        activate DB
        DB-->>Ctrl: Sukses Simpan Tiket
        deactivate DB
        Ctrl-->>Part: Redirect ke Dashboard Tiket Participant
    end
    deactivate Ctrl
