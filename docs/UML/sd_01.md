sequenceDiagram
    autonumber
    actor User as Calon Pengguna
    participant View as Register Blade View
    participant Ctrl as RegisteredUserController
    participant DB as MongoDB (users)

    User->>View: Isi Nama, Email, Sandi, & Pilih Role
    User->>View: Klik Tombol "Register"
    activate View
    View->>Ctrl: HTTP POST /register (Form Data)
    deactivate View
    activate Ctrl
    Ctrl->>Ctrl: Validasi Input (Nama, Email, Role)
    alt Data Tidak Valid
        Ctrl-->>User: Redirect Back dengan Pesan Error
    else Data Valid
        Ctrl->>DB: Insert Dokumen Baru (role: organizer/participant)
        activate DB
        DB-->>Ctrl: Sukses Simpan & Kembalikan _id
        deactivate DB
        Ctrl->>Ctrl: Set Web Session Auth (Login Otomatis)
        Ctrl-->>User: Redirect ke Dashboard Sesuai Role
    end
    deactivate Ctrl
